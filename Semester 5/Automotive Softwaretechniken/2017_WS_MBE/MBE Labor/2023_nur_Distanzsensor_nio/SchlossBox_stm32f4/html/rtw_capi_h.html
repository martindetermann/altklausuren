149/* DataTypeMap structure */



150/* Members provide data type information of a signal or parameter */ 



151typedef struct rtwCAPI_DataTypeMap_tag {



152  const char_T *cDataName;   /* C language data type name                    */



153  const char_T *mwDataName;  /* MathWorks data type, typedef in rtwtypes.h   */



154  uint16_T      numElements; /* number of elements, 0 for non-structure data */



155  uint16_T      elemMapIndex;/* index into the ElementMap, gives Bus Info    */



156  uint16_T      dataSize;    /* data size in Bytes                           */



157  uint8_T       slDataId;    /* enumerated data type from simstruc_types.h   */



158  unsigned int  isComplex:1; /* is the data type complex (1=Complex, 0=Real) */



159  unsigned int  isPointer:1; /* is data accessed Via Pointer (1=yes, 0= no)  */



160} rtwCAPI_DataTypeMap;



161



162/* Notes on rtwCAPI_DataTypeMap:



163 * cDataName  - The ANSI C equivalent data type. 



164 *              For fixed point data



165 *                  cDataName = the native integer equivalent



166 *              For Complex data or non-virtual structures, 



167 *                  cDataName = "struct" 



168 * mwDataName - RTW defined data type. Typedef can be found in rtwtypes.h



169 *              For fixed point data, 



170 *                  mwDataName = the integer type in which the data is stored.



171 *                  Use rtwCAPI_FixPtMap to get slope & bias of the data.



172 * numElements - Number of elements/members in the data type. If the data type 



173 *              represents a non-virtual structure (for e.g, bus structure),



174 *              this field gives the number of elements in the structure. 



175 *              For non-structure data type, the default value is 0



176 *              Complex data types are treated as non-structures



177 * elemMapIndex - Index into the rtwCAPI_ElementMap which maps each of the



178 *              element/member of a structure data type.



179 * slDataId   - Enumerated Simulink Data type defined in simstruc_types.h.



180 *              The enumerations are



181 *                 o SS_DOUBLE



182 *                 o SS_SINGLE



183 *                 o SS_INT8



184 *                 o SS_UINT8



185 *                 o SS_INT16



186 *                 o SS_UINT16



187 *                 o SS_INT32



188 *                 o SS_UINT32



189 *                 o SS_BOOLEAN



190 *                 o SS_ENUM_TYPE  (for enumerated data types)



191 *                 o SS_STRUCT     (for non-virtual structures)



192 *              For fixed point data: 



193 *                  slDataId = enumeration corresponding to the integer type 



194 *                             in which the data is stored



195 *              For Complex data: 



196 *                  slDataId = enumeration corresponding to the data type of 



197 *                             the real part.



198 * isComplex  - is the data type complex (1=Complex, 0=Real) 



199 * isPointer  - is data accessed Via Pointer (1=yes, 0= no)



200 */



201

226/*



227 * DECL_AND_INIT_DIMSINFO(variableName):



228 *    Macro for setting up a DimsInfo in an S-function to DYNAMIC_DIMENSION.



229 *    This macro must be placed at the start of a declaration, for example:



230 *



231 *           static void mdlInitializeSizes(SimStruct *S)



232 *           {



233 *               DECL_AND_INIT_DIMSINFO(diminfo1);



234 *



235 *               ssSetNumSFcnParams(S, 0);



236 *               <snip>



237 *           }



238 *



239 *    The reason that this macro must be placed in the declaration section of a



240 *    function or other scope is that this macro **creates** a local variable of



241 *    the specified name with type DimsInfo_T. The variable is initialized



242 *    to DYNAMIC_DIMENSION.



243 */



244#define DECL_AND_INIT_DIMSINFO(variableName)    \

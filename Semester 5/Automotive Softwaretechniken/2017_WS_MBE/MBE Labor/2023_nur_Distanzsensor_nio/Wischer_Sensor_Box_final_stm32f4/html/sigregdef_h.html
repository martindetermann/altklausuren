25#define SLREG_NO_FLAGS (0)

27#define SLREG_AVAIL           (1U << 0) 



28#define SLREG_REUSED          (1U << 1)



29#define SLREG_FCNCALL         (1U << 2)



30#define SLREG_MERGE           (1U << 3)



31#define SLREG_GROUND          (1U << 4) 



32#define SLREG_FRAME           (1U << 5)  /* opt exclude flag-see below          */



33#define SLREG_WIDE_FRAME      (1U << 6)  /* opt exclude flag-see below          */



34#define SLREG_MATRIX          (1U << 7)  /* opt exclude flag-see below          */



35#define SLREG_COMPLEX         (1U << 8)  /* opt exclude flag-see below          */



36#define SLREG_ALLOW_REUSED    (1U << 9)  /* opt un-exclude flag - see below     */



37#define SLREG_ACTION          (1U << 10) 



38#define SLREG_CONDEXEC        (1U << 11) /* signal is only conditional executed */



39#define SLREG_EXTMODE_UNAVAIL (1U << 12) /* signal is not available in extmode  */



40#define SLREG_ELIMINATED      (1U << 13) /* dead branch elimination             */



41#define SLREG_INCLUDE_GRND    (1U << 14) /* opt include ground regions          */



42#define SLREG_VIRTSIGCNV      (1U << 15) /* signal connected to signal type 



43                                          * conversion such as frame conversion



44                                          */



45#define SLREG_VARDIMS         (1U << 16) 



46#define SLREG_MESSAGES        (1U << 17)



47#define SLREG_DES_STAT        (1U << 18)

49#define IS_BIT_SET(uword32,bit) ((uword32 & bit) != 0)

51/* region node */



52typedef struct SL_SigRegion_tag {



53#if !defined(S_FUNCTION_NAME) && !defined(BUILDING_SL_LOGGING)



54    slPort *portObj;        /* output port associated with reg */



55#else



56    void *portObj;



57#endif



58    void         *data;     /* pointer to the data              */



59    int          startIdx;  /* starting element of the region (i.e., 'data'



60                             * points to the 'startIdx' element of the port)



61                             */



62    int          nEls;      /* number of elements in region     */



63    bool         entireSig; /* is this region the whole sig?    */



64    unsigned int status;    /* bit field: see SLREG_* status



65                             * vals above */



66



67    int index; /* The i'th region in the original map.  So, if a mux



68                * has three sigs coming in, a,b,c, a is index 0, b 1,



69                * and c 3.  If the regions were an array instead of



70                * a linked list this would be the array index. 



71                * BE CAREFUL: If regions are removed due to being



72                * duplicate or unavailable, the index field is not



73                * touched up, so the index field always remains what



74                * the original signal map for the composite signal



75                * represented.  I'm not sure if this is good behavior



76                * or not, but this is how it currently works:



77                * 1/16/04 HJT



78                */



79



80    int workInt;  /* A work integer.  Temp mem location for use when



81                   * shuffling regions.



82                   */



83



84    /*



85     * The following can be derived from the port, but since s-functions



86     * do not currently have direct access to the internal slPort *,



87     * we must put a copy of the required info directly into this struct.



88     */



89    int dType;



90    int dTypeSize;



91    int complexity;



92    int frameSize;



93



94    /*



95     *



96     */



97    bool discrete;   /*true if discrete in time or value (draw as stair)*/



98



99    /*



100     * Dimensions



101     *



102     * o Notes:



103     *    If not a frame (frameSize == 0)



104     *       dims[0...numDims], numDims = 1 for a scalar signal



105     *    If frame 



106     *       dims[0] = frameSize



107     *       dims[1] = number of columns



108     *    basically, the frame is treated as a matrix.



109     *



110    */



111    int numDims;



112    int *dims;



113



114    /* Cache var dims info in the sig region */



115    int     *currDims; 



116



117    struct SL_SigRegion_tag *next;



118    struct SL_SigRegion_tag *prev;



119



120    const char *name;



121



122    int unitId;



123    



124} SL_SigRegion;

127/*******************************************************************************



128 *                  SIGREGION: PUBLIC ACCESSORS                                *



129 ******************************************************************************/



130



131/*



132 * The number of elements in the region.



133 */



134#define gsr_nEls(reg) ((reg)->nEls)



135



136/*



137 * A pointer to the next region in the linked list. NULL if at the end of



138 * the list.



139 */



140#define gsr_NextReg(reg) ((reg)->next)



141



142/*



143 * A const char * pointer into the data.



144 */



145#define gsr_data(reg) ((char *)((reg)->data))



146



147/*



148 * Starting element of the region (i.e., 'data' points to the 'startIdx'



149 * element of the port)



150 */



151#define gsr_startIdx(reg) ((reg)->startIdx)



152



153/*



154 * Output port associated with region.



155 */



156#define gsr_portObj(reg) ((reg)->portObj)



157



158/* 



159 * The simulink data type identifier for the region.



160 */



161#define gsr_DataType(reg) ((reg)->dType)



162



163/*



164 * The number of bytes in one, non-complex data value.



165 */



166#define gsr_DataTypeSize(reg) ((reg)->dTypeSize)



167



168/*



169 * The simulink unit identifier for the region.



170 */



171#define gsr_Unit(reg) ((reg)->unitId)



172



173/*



174 * True if the data region contains complex elements.



175 */



176#define gsr_Complex(reg) ((reg)->complexity)



177



178/*



179 * The number of rows.  If not matrix, -1.



180 * NOTE: 



181 *  This access macro is deprecated and will only work correctly if 



182 *  gsr_NumDims <= 2 (i.e.: this macro does not support N-D signals N >2)



183 */



184#define gsr_M(reg) (((reg)->numDims <=1) ? -1 : (reg)->dims[0])



185



186/*



187 * The number of cols.  If not matrix, the sig width of the corresponding



188 * region. 



189 * NOTE: 



190 *  This access macro is deprecated and will only work correctly if 



191 *  gsr_NumDims <= 2 (i.e.: this macro does not support N-D signals N >2)



192 */



193#define gsr_N(reg) (((reg)->numDims <=1) ? (reg)->nEls : (reg)->dims[1])



194



195/*



196 * The number of dimensions. Can be greater than 2 for N-dimensional signals.



197 */



198#define gsr_NumDims(reg) ((reg)->numDims)



199



200/*



201 * An integer pointer to the dimensions vector. Use gsr_NumDims to determine



202 * the length of the dimensions array.



203 */



204#define gsr_Dims(reg) ((reg)->dims)



205



206/*



207 * The frameSize of the signal. Zero if not frame.



208 */



209#define gsr_FrameSize(reg) ((reg)->frameSize)



210



211/*



212 * The availability status of the region (SLREG_AVAIL if avail, else one



213 * of the status described above in the "Public Description" section.



214 */



215#define gsr_status(reg) ((reg)->status)



216



217#define gsr_index(reg) ((reg)->index)



218



219/*



220 * Var dims info - int pointer to current dims



221 */



222#define gsr_currDims(reg) ((reg)->currDims)



223



224



225#endif /* sigregdef_h */

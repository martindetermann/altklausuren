9#ifndef _DTINFO_H_

11



12/* Data type transition */



13typedef struct DataTypeTransition_tag {



14  char_T *baseAddr; /* starting address of the transition  */



15  int_T  dataType;



16  int_T  isComplex; /* elements of this region are complex */



17  int_T  nEls;



18} DataTypeTransition;



19



20/* Data type transition table */



21typedef struct DataTypeTransitionTable_tag {



22  uint_T             numTransitions;     /* number of transitions in table  */



23  DataTypeTransition *transitions;       /* base address of transition table*/



24} DataTypeTransitionTable;



25



26/*



27 * The model's data type transition info structure is attached



28 * to the SimStruct:



29 *



30 * DataTypeTransInfo *dtInfo = ssGetModelMappingInfo(S);



31 *



32 * Additional information is accessed via:



33 * 



34 * uint_T NumDataTypes     = dtGetNumDataTypes(dtInfo);



35 * uint_T DataTypeSizes[]  = dtGetDataTypeSizes(dtInfo);



36 * char_T *DataTypeNames[] = dtGetDataTypeNames(dtInfo);



37 *



38 * DataTypeTransTable *B     = dtGetBIODataTypeTrans(dtInfo);



39 * DataTypeTransTable *P     = dtGetParamDataTypeTrans(dtInfo);



40 * DataTypeTransTable *DWork = dtGetDWorkDataTypeTrans(dtInfo);



41 * DataTypeTransTable *Xd    = dtGetDiscStatesDataTypeTrans(dtInfo);



42 * DataTypeTransTable *U     = dtGetExternalInputsDataTypeTrans(dtInfo);



43 * DataTypeTransTable *Y     = dtGetExternalOutputsDataTypeTrans(dtInfo);



44 *



45 * And, using the block outputs as an example



46 * 



47 * uint_T              NumBIOTransitions  = dtGetNumTransitions(B);



48 * DataTypeTransition  *BIOTransitions    = dtGetTransitions(B);



49 * char_T              *address           = dtTransGetAddress(B, idx);



50 * int_T               dataType           = dtTransGetDataType(B, idx);



51 * int_T               isComplex          = dtTransGetComplexFlag(B, idx);



52 *



53 * where, idx is the index into the block outputs transition table.



54 * For example, dtTransGetComplexFlag(B, 5) indicates whether the fifth



55 * transition represents a complex region of the block outputs structure



56 * structure and dtTransGetAddress(B, 5) is the absolute base address of



57 * the region.



58 */



59 



60typedef struct DataTypeTransInfo_tag {



61  uint_T         numDataTypes;    /* number of data types in model    */



62  uint_T         *dataTypeSizes;  /* data types (bytes) of data types */



63  char_T  const **dataTypeNames;  /* names of data types              */



64



65  DataTypeTransitionTable *BTransTable;     /* block outputs */



66  DataTypeTransitionTable *PTransTable;     /* parameters */



67  DataTypeTransitionTable *DWorkTransTable; /* data type work vector */



68  DataTypeTransitionTable *XdTransTable;    /* discrete states */



69  DataTypeTransitionTable *UTransTable;     /* model inputs */



70  DataTypeTransitionTable *YTransTable;     /* model outputs */



71} DataTypeTransInfo;



72



73#define dtGetNumDataTypes(dtInfo)   ((uint_T)((dtInfo)->numDataTypes))



74#define dtGetDataTypeSizes(dtInfo)  ((dtInfo)->dataTypeSizes)



75#define dtGetDataTypeNames(dtInfo)  ((dtInfo)->dataTypeNames)



76



77#define dtGetBIODataTypeTrans(dtInfo)             ((dtInfo)->BTransTable)



78#define dtGetParamDataTypeTrans(dtInfo)           ((dtInfo)->PTransTable)



79#define dtGetDWorkDataTypeTrans(dtInfo)           ((dtInfo)->DWorkTransTable)



80#define dtGetDiscStatesDataTypeTrans(dtInfo)      ((dtInfo)->XdTransTable)



81#define dtGetExternalInputsDataTypeTrans(dtInfo)  ((dtInfo)->UTransTable)



82#define dtGetExternalOutputsDataTypeTrans(dtInfo) ((dtInfo)->YTransTable)



83



84#define dtGetNumTransitions(dtTable) \



85    ((dtTable)->numTransitions)

87#define dtGetTransitions(dtTable) \

95



96#define dtTransNEls(dtTable, idx) \

98



99#define dtTransGetComplexFlag(dtTable, idx) \



100    ((dtTable)->transitions[(idx)].isComplex)
